 Here is the SQL query which is able to build for example a project list with one row per category linked to a project:

-- Request to retrieve a detailed list of projects with one row per category attached to project
SELECT udb.name, udb.fullname, phm.name, udb.startdate, prj.end_date, prj.close_date, cge.label as category
FROM project prj
    INNER JOIN userdatabase udb on (udb.databaseid = prj.databaseid)
    INNER JOIN phase pha on prj.id_current_phase = pha.id_phase
    INNER JOIN project_model prm on prj.id_project_model = prm.id_project_model
    INNER JOIN phase_model phm on phm.id_project_model = prm.id_project_model
    INNER JOIN project_details pdt on pdt.id_project_model = prm.id_project_model
    INNER JOIN layout lay on (lay.id_layout = phm.id_layout or lay.id_layout = pdt.id_layout)
    INNER JOIN layout_group lyg on lyg.id_layout = lay.id_layout
    INNER JOIN layout_constraint lyc on lyc.id_layout_group = lyg.id_layout_group
    INNER JOIN flexible_element fle on fle.id_flexible_element = lyc.id_flexible_element
    LEFT OUTER JOIN question_element que on fle.id_flexible_element = que.id_flexible_element
    LEFT OUTER JOIN category_type cgt on que.id_category_type = cgt.id_category_type
    LEFT OUTER JOIN category_element cge on cgt.id_category_type = cge.id_category_type
    LEFT OUTER JOIN value val on (val.id_project = prj.databaseid and val.id_flexible_element = fle.id_flexible_element)
    INNER JOIN question_choice_element qce on (que.id_flexible_element = qce.id_question and cge.id_category_element = qce.id_category_element and ARRAY[qce.id_choice] <@ string_to_array(val.value, '~')::bigint[])
WHERE qce.id_question is not null;
